  { uid: 'user_6', name: 'User 6', age: 95 },
  { uid: 'user_7', name: 'User 7', age: 49 },
  { uid: 'user_8', name: 'User 8', age: 45 },
  { uid: 'user_9', name: 'User 9', age: 29 },

## Installation

```
npm install dummy-data-maker
```
